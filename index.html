Ejercicio 3 EXISTS con relación real 
-- Buscar alumnos que tengan calificación mayor a 90
SELECT * FROM Alumnos a
WHERE ___ (
    SELECT 1 FROM Inscripciones i
    ___ Grupos g ON i.GrupoID = g.GrupoID
    WHERE i.AlumnoID = a.AlumnoID 
    ___ i.CalificacionFinal > 90
)
Ejercicio 4 Analizando resultados 
SELECT a.Matricula, a.Nombre, COUNT(i.InscripcionID) as TotalMaterias
FROM Alumnos a
LEFT JOIN Inscripciones i ON a.AlumnoID = i.AlumnoID
GROUP BY a.Matricula, a.Nombre
HAVING COUNT(i.InscripcionID) > 2
